#!/bin/bash

# Good Date Format 

date '+%A - %d %B %Y - %T'


# Example
echo $(date '+%A - %d %B %Y - %T')


######################### Format for scripts ###########################
#Todays date:
date -I
 
#Yesterdays date:
date -I -d "1 day ago"

#This Hour and day:
date +%Y-%m-%d-%p-%H

#Previous Hour and this day
date +%Y-%m-%d-%p-%H -d "1 hour ago"

#This Minute , Hour and day:
date +%Y-%m-%d-%p-%H-%M

#Previous Minute , this Hour and day:
date +%Y-%m-%d-%p-%H-%M -d "1 minute ago"

# This day,hour,minute & second
date +%Y-%m-%d-%p-%H-%M-%S

#29 days ago in ISO-8601 format
date -I -d "29 days ago"
