#!/bin/bash
# How to find duplicates in Original


# My Way
{

  dupes=`find /media/Original/ |sort -bif| sed 's/.mpeg//' | sed 's/.mpg//' |sed 's/.mkv//' | sed 's/.avi//' |sed 's/.mp4//' |uniq -id > ~/.sed-buffer` ; cat ~/.sed-buffer |echo -e "\n\n\nThere are `wc -l` duplicate episodes\n\n\n"  ; cat ~/.sed-buffer |  sed 's/ /\\ /g' | sed "s/'\+/\\\'/g" | sed "s/(/\\\(/g" | sed "s/)/\\\)/g"

}

# Internet way
{
  
  find /media/Original/ -exec basename {} \; | sed 's/\(.*\)\..*/\1/' | sort | uniq -c | grep -v "^[ \t]*1 "

}
