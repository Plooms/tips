#!/bin/bash
# Use the (screen) command to control the terminal

# Install

  p screen
  
  # Optional (to attach a running program to a screen session)
  p reptyr



  
# Commands are entered pressing ctrl+a and then the key binding.


# Common Commands
{

    ctrl+a ? 		"Displays commands and its defaults (VERY important)"
    
    ctrl+a : 		"Enter to the command prompt of screen"
    
    ctrl+a " 		"Window list""
    
    ctrl+a 0 		"opens window 0"
    
    ctrl+a A 		"Rename the current window"
    
    ctrl+a a 		"Sends ctrl+a to the current window"
    
    ctrl+a c 		"Create a new window (with shell)"
    
    ctrl+a S 		"Split current region into two regions"
    
    ctrl+a Esc 		"Enter Copy Mode (use enter to select a range of text)"
    
    ctrl+a ] 		"Paste text"
    
    ctrl+a Q 		"Close all regions but the current one"
    
    ctrl+a X 		"Close the current region"
    
    ctrl+a d 		"Detach from the session, and leave it running. Use screen -r to resume"

}


  # The escape key can be changed with the escape option in ~/.screenrc. For example:


  escape ``


  # sets the escape key to `.
    
        
# If you started a program outside screen, but now you would like it to be inside,
# you can use reptyr to reparent the process from its current tty to one inside screen. 
 
 # NOTE: to get PID of a process 
 ps aux

 reptyr <pid>
