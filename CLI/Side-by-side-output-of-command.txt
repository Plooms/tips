#!/bin/bash
# Paste the output of two files side by side




# To try it, 
#(this prints a number from the first file then from the second, etc.)

for i in {1..10};do echo $i;done >| /tmp/num1

for i in {11..20};do echo $i;done >| /tmp/num2

while read num1 num2 ;do echo $num1 $num2;done < <(paste /tmp/num1 /tmp/num2)
