#!/bin/bash
# How to see duplicates in your media


# copy pasta all this in terminal

find NAS/Shows -iname "*srt*duplicate*mpeg" -size -2M -delete -print
find NAS/Shows -iname "*duplicate*mpeg" \
| cut -d"]" -f1 \
| rev \
| cut -d"/" -f1 \
| rev \
| while read a ; do find NAS/Shows -iname "$a*" -size +10M -exec du -hs {} \; ; echo "" ;done
