#!/bin/bash

# NOTE : you can restore old settings from .config/vlc #


# To add speed bar to player toolbar
Customize interface > then just drag the speed applet to the fullscreen toolbar

# To pause when minimized
preferences > interface > Pause playback when minimized

# To make audio louder
preferences > show all settings > Audio > Audio gain (2.00)

# To make english the default language
preferences > show all settings > Input/Codecs > <Audio|Subtitle> Language : en,eng,us

# To make seeking faster
preferences > show all settings > Input/Codecs > Fast Seek 

# To change the buffer size
preferences > show all settings > Input/Codecs >  <File|Network> Caching (ms)

# To always display the filename instead of the metadata
preferences > show all settings > Input/Codecs > Change title according to current media = $

# To start in fullscreen
preferences > video > fullscreen

# To hide (player bar & mouse) faster
preferences > show all settings > Video > Hide cursor and fullscreen (1 ms)

# To change Jump length
preferences > show all settings > Hotkey setttings > Jump sizes (3,30,60,9999)

# To add mouse gestures
preferences > show all settings > Control interfaces > mouse gestures

# To show video title on video start 
preferences > show all settings > Video > Show media title + show video title for X ms

# To force aspect ratio
preferences > show all settings > Video > source aspect ratio = 16:9

# To quit vlc after playback
n .xbmc/userdata/playercorefactory.xml 


	<args>"{1}".xspf --play-and-exit </args>





# Mouse Gestures
<     left 			: (Short time) skip backward (10sec by default)
>     right 		: (Short time) skip forward (10sec by default)
< ^   left-up		: Faster
> ^   right-up		: Slower
< v   left-down 	: Go to previous entry in playlist
> v   right-down 	: Go to next entry in playlist
< >   left-right 	: Play/Pause
> <   right-left 	: Play/Pause 

^     up 			: Volume up
v     down 			: Volume down
^ v   up-down	 	: Mute Volume
v ^   down-up 		: Mute Volume
^ >   up-right 		: Change Audio track
v >   down-right	: Change Subtitle track
^ <   up-left 		: Enter fullscreen mode
v <   down-left 	: Quit VLC 
# To disable audio and get it back do:
down-right - up-right
