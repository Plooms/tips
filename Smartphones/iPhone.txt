#!/bin/bash
# Mounting iPhone on Arch

# Install packages
    sudo pacman -S ifuse usbmuxd libplist

# Load Module
    sudo modprobe fuse



#NOTE: Now Unlock the device to gain access through the USB interface

# Mount the device
     sudo ifuse /yada/yada/path-to-mountfolder

# Unmount after finishing
     sudo umount /yada/yada/path-to-mountfolder
