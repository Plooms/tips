#!/bin/bash


# Change the I/O Scheduler
{  
  
  su -c 'echo bfq > /sys/block/sda/queue/scheduler'

}

# Check the available schedulers
{

  cat /sys/block/sda/queue/scheduler

}

# Set the Scheduler permanently
{

  sudo nano /etc/tmpfiles.d/IO_scheduler.conf

w /sys/block/sda/queue/scheduler - - - - bfq
w /sys/block/sdb/queue/scheduler - - - - bfq
  
}
