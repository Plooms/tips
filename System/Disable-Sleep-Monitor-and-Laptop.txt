#!/bin/bash


# Disable sleep when closing Laptop lid
{
  
  n /etc/systemd/logind.conf
   
   
  # Edit
  HandleLidSwitch=ignore

  
  # then execute
  sudo systemctl restart systemd-logind
  
}


# Disable Monitor screen from blanking or turning off
{

  sudo xset -dpms; sudo xset s off

}

# Turn off screen now
{
	sleep 1; xset dpms force off
	
	or
	
	sleep 1; xset dpms force standby
	
	or
	
	sleep 1; xset dpms force suspend
}
