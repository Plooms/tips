#!/bin/bash
# Compress programs to make them faster



 p upx


# First , make a backup of the binary:
{

 cp /usr/lib/firefox/firefox /usr/lib/firefox/firefox.backup

}

# Compress
{

 upx --best /usr/lib/firefox/firefox

}

# To uncompress (haha)
{

 mv  /usr/lib/firefox/firefox.backup  /usr/lib/firefox/firefox
 
}
