#!/bin/bash
# SMART info


	# Install
	p smartmontools
 
 
	# Test
	sudo smartctl -t short /dev/xxx
	or
	sudo smartctl -t long /dev/xxx

	# Results
	sudo smartctl -H /dev/xxx		# Errors

	sudo smartctl -a /dev/xxx		# General Info
	
	
	
  # Find Failed HDD to remove
  
  p smartmontools

  # To find the serial number which is printed on the face of the HDD or by scanning the barcode

  sudo smartctl -i /dev/sdXX
