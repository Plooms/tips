#!/bin/bash
# Install X2go (the alternative to VNC)



# Install
 p x2goserver x2goclient


# Create the database
 sudo x2godbadmin --createdb

# Start the service in all servers
 sudo systemctl restart x2goserver.service
 sudo systemctl enable x2goserver.service

# Start the client
 x2goclient
